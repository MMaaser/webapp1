const express = require('express') // imports express library for server stuff 
const app = express() // creates server
const port = 3000

app.get('/', (req, res) => { //when someone visits the root page, run this function with parameters srequeset and response
res.send('index.html')
})

app.get('/edit', (req, res) => {
    res.send('edit')
})
//starts the server
app.listen(port, () => {
    console.log(`Example app listening on port ${port}`) // use the tilde for variables within strings
})
